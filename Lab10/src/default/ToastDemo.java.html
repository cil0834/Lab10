public class ToastDemo extends JFrame 
implements ChangeListener{

	private JButton b1;
	private JButton b2;
	private JTextArea text;
	private JSlider slider;
	private Font f32 = new Font("bigfont", Font.BOLD, 32);
	private JList<String> list;
	
	public ToastDemo()
	{
		initFrame();
		
		setButtons();
		
		setSlider();
		
		setText();
		
		//setList();
		
		add(b1, BorderLayout.NORTH);
		add(b2, BorderLayout.WEST);
		add(text, BorderLayout.CENTER);
		//add(list, BorderLayout.EAST);
		add(slider, BorderLayout.SOUTH);
		setVisible(true);
	}

	
	private void setButtons()
	{
		b1 = new JButton("b1");
		b1.setFont(f32);
		b2 = new JButton("b2");
		b2.setFont(f32);
		
		b1.addActionListener(new ActionListener() {
			
			@Override
			public void actionPerformed(ActionEvent e)
			{
				int ret = JOptionPane.showConfirmDialog(ToastDemo.this, "Do you want to check it?",
						"yes, I do", JOptionPane.YES_NO_CANCEL_OPTION,
						JOptionPane.QUESTION_MESSAGE);
				text.setText("Option " + ret);
			}
			
			
		});
		
		b2.addActionListener(new ActionListener() {
			
			@Override
			public void actionPerformed(ActionEvent e)
			{
				Color color = JColorChooser.showDialog(ToastDemo.this,
						"Choose Backgroun Color", b2.getBackground());
				
				b2.setBackground(color);
			}
			
			
		});
		
		
		
	}

	{
		setSize(700, 400);
		this.setLocationRelativeTo(null);
		this.setDefaultCloseOperation(EXIT_ON_CLOSE);
		BorderLayout layout = new BorderLayout();
		this.setLayout(layout);
		//this.setVisible(true);
		
	}
	
	private void setText()
	{
		text = new JTextArea();
		text.setLineWrap(true);
		text.setWrapStyleWord(true);
		text.setFont(f32);
	}
